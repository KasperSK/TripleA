\subsection{Use Case 2}
I denne Use Case beskrives forløbet at udskrive en bon, hvis dette ønskes.
\begin{usecase}{2}

\title{Udskriv Kundebon} 

\field{Mål:}{ Systemet får udskrevet en bon af sidst gennemførte salg }

\field{Initieret af:}{ Bartender }

\field{Aktører:}{ Bartender (Primær) }

\field{Samtidige forekomster:}{1}

%Preconditions: What must be true on start and worth telling the reader?
\field{Prækondition:}{Use Case 1: Kontant salg er gennemført}

%Postconditions: What must be true on successful completion and worth telling the reader
\field{Postkondition:}{Systemet udskriver bon af sidst gennemførte salg }

%Main Success Scenario: A typical, unconditional happy path scenario of success.
\scenario{Hovedscenarie:}{
	\item Bartender trykker på knappen der printer bon
	\item Systemet udskriver bon 
	 \newline
	[Extension 2.1: Der er ikke mere papir]
}

%Extensions: Alternate scenarios of success or failure.
\scenario{Udvidelser:}{
	\item Extension 2.1 Der er ikke mere papir
		\begin{enumerate}
		\item Systemet printer ikke bon
		\end{enumerate}
}
\end{usecase}
